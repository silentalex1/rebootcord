import os
import discord
from discord.ext import commands
import random
import string
import urllib.request
import json

try:
    from dotenv import load_dotenv
    load_dotenv()
except (ImportError, AttributeError):
    pass

TOKEN = os.getenv("BOT_TOKEN")

if not TOKEN:
    print("ERROR: BOT_TOKEN not found in environment.")
    exit(1)

OWNERS = [1300548891611234356]

def gen_invite():
    a = ''.join(random.choices(string.ascii_lowercase + string.digits, k=5))
    b = ''.join(random.choices(string.ascii_lowercase + string.digits, k=7))
    return f"rebootcord-{a}-{b}"

intents = discord.Intents.default()
intents.message_content = True
intents.voice_states = True

bot = commands.Bot(command_prefix="$", intents=intents)

@bot.event
async def on_ready():
    print(f"Logged in as {bot.user} (ID: {bot.user.id})")
    print("Reboot Cord bot is now online locally.")
    print("Ready to receive commands.")

@bot.command()
async def ping(ctx):
    await ctx.send("Pong! Bot is running locally.")

@bot.command()
async def createcode(ctx, member: discord.Member = None):
    if ctx.author.id not in OWNERS:
        await ctx.send("You do not have permission to use this command.")
        return
    if member is None:
        await ctx.send("Please mention a user to create the code for: $createcode @user")
        return
    discord_user = member.name
    code = gen_invite()
    payload = {"code": code, "user": discord_user}
    try:
        data_bytes = json.dumps(payload).encode("utf-8")
        req = urllib.request.Request(
            "https://rebootcord.onrender.com/api/createcode",
            data=data_bytes,
            headers={"Content-Type": "application/json"}
        )
        with urllib.request.urlopen(req, timeout=10) as response:
            resp_data = json.loads(response.read().decode("utf-8"))
            if resp_data.get("success"):
                await ctx.send(f"invite code created and generated for {member.mention}")
                try:
                    await member.send(f"Your Reboot Cord invite code is: {code}\n\nThis code is bound to your Discord username {discord_user}.\nUse it to register at https://rebootcord.onrender.com/")
                except:
                    await ctx.send("(Could not DM the user — they may have DMs disabled from this bot.)")
            else:
                await ctx.send("Failed to create code: " + str(resp_data))
    except Exception as e:
        await ctx.send("Error creating code: " + str(e))

if __name__ == "__main__":
    print("Starting Reboot Cord Discord bot locally...")
    bot.run(TOKEN)
